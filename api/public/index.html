Public Key API Endpoints

  • POST /pk/user - Register a new user with their public key. 
    curl -X POST /pk/user \
    -H "Content-Type: application/json" \
    -d '{"user": "client|NewUser01"}'

NFT API Endpoints

  • POST /assets/createTokenClass - Create a new token class. 
    curl -X POST /assets/createTokenClass \
    -H "Content-Type: application/json" \
    -d '{
    "network": "GC",
     "tokenClass": {
    "collection": "Membership",
    "category": "Employer",
    "type": "Standard",
    "additionalKey": "Epic"
    },
    "isNonFungible": true,
    "decimals": 0,
    "name": "Standard Membership",
    "symbol": "MBRE",
    "description": "A detailed description of the token.",
    "rarity": "Epic",
    "image": "https://example.com/image.png",
    "metadataAddress": "https://example.com/metadata.json",
    "contractAddress": "none",
    "maxSupply": "10000",
    "maxCapacity": "10000",
    "totalMintAllowance": "5000000",
    "totalSupply": "0",
    "totalBurned": "0"
    }'
  • POST /assets/grantAllowance - Gives authority to mint token instances. 
    curl -X POST /assets/grantAllowance \
    -H "Content-Type: application/json" \
    -d '{
    "collection": "Membership",
    "category": "Employer",
    "type": "Standard",
    "additionalKey": "Epic",
    "allowanceType": 4,
    "quantities": [{
    "user": "client|NewUser01",
    "quantity": "1"
    }],
    "uses": "5"
    }'
  • POST /assets/mintToken - Mints token instances. 
    curl -X POST /assets/mintToken \
    -H "Content-Type: application/json" \
    -d '{
    "tokenClassKey": {
    "collection": "Membership",
    "category": "Employer",
    "type": "Standard",
    "additionalKey": "Epic"
    },
    "owner": "client|NewUser01",
    "quantity": "1",
    "userPrivateKey": "private-key-here"
    }'
  • POST /assets/burnTokens - Burns token instances. 
    curl -X POST /assets/burnTokens \
    -H "Content-Type: application/json" \
    -d '{
    "tokenInstances": [
    {
    "quantity": "1",
    "tokenInstanceKey": {
    "collection": "Membership",
    "category": "Employer",
    "type": "Standard",
    "additionalKey": "Epic",
    "instance": "1"
    }
    }
    ],
    "owner": "client|NewUser01",
    "userPrivateKey": "private-key"
    }'
  • GET /assets/getAllowances - Retrieve the token allowances for a user. 
    curl -X GET "/assets/getAllowances?tokenInstanceKey=Membership$Employer$Standard$Epic$0&user=client|admin&grantedTo=client|NewUser01" \
    -H "Content-Type: application/json"
  • POST /assets/lockToken - Lock a token instance. 
    curl -X POST /assets/lockToken \
    -H "Content-Type: application/json" \
    -d '{
    "owner": "client|NewUser01",
    "lockAuthority": "client|NewUser01",
    "tokenInstance": {
    "collection": "Membership",
    "category": "Employer",
    "type": "Standard",
    "additionalKey": "Epic",
    "instance": "1"
    },
    "quantity": "1",
    "useAllowances": ["5"],
    "userPrivateKey": "private-key"
    }'
  • POST /assets/unlockToken - Unlock a token instance. 
    curl -X POST /assets/unlockToken \
    -H "Content-Type: application/json" \
    -d '{
    "owner": "client|NewUser01",
    "tokenInstance": {
    "collection": "Membership",
    "category": "Employer",
    "type": "Standard",
    "additionalKey": "Epic",
    "instance": "1"
    },
    "quantity": "1",
    "lockedHoldName": "hold-name",
    "userPrivateKey": "private-key"
    }'
